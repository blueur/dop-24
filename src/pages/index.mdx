## Calendrier

import React from "react";
import Calendar from "./_calendar.mdx";

export class CurrentCalendar extends React.Component {
  componentDidMount() {
    const date = new Date();
    const day = date.getDay();
    const currentDate = new Date(
      date.setDate(date.getDate() - day + (day === 0 ? -2 : 5))
    );
    const dateText = currentDate.toISOString().split("T")[0];
    const weekend = day === 0 || day === 6;
    Array.from(document.querySelectorAll("td"))
      .filter((a) => a.textContent === dateText)
      .forEach((element) => {
        const parent = element.parentElement;
        parent.classList.add("current", weekend ? "weekend" : "week");
        parent.scrollIntoView({ behavior: "smooth" });
      });
  }
  render() {
    return <Calendar />;
  }
}
